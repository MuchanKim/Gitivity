**Tech Stack:** Swift 6, SwiftUI, FoundationModels, AuthenticationServices, GitHub GraphQL/REST API, Cloudflare Workers (Wrangler CLI)

```swift
import Foundation
import Security

struct KeychainService {
    private let serviceName = "io.moolab.gitivity"

    func save(key: String, value: String) throws {
        let data = Data(value.utf8)
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: serviceName,
            kSecAttrAccount as String: key,
            kSecValueData as String: data
        ]
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw KeychainError.saveFailed(status)
        }
    }

    func read(key: String) throws -> String? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: serviceName,
            kSecAttrAccount as String: key,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess, let data = result as? Data else {
            if status == errSecItemNotFound { return nil }
            throw KeychainError.readFailed(status)
        }
        return String(data: data, encoding: .utf8)
    }

    func delete(key: String) throws {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: serviceName,
            kSecAttrAccount as String: key
        ]
        let status = SecItemDelete(query as CFDictionary)
        guard status == errSecSuccess || status == errSecItemNotFound else {
            throw KeychainError.deleteFailed(status)
        }
    }
}

enum KeychainError: Error {
    case saveFailed(OSStatus)
    case readFailed(OSStatus)
    case deleteFailed(OSStatus)
}
```

```typescript
export interface Env {
  GITHUB_CLIENT_ID: string;
  GITHUB_CLIENT_SECRET: string;
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url);

    if (url.pathname === "/token" && request.method === "POST") {
      const body = await request.json<{ code: string }>();

      const response = await fetch("https://github.com/login/oauth/access_token", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Accept": "application/json",
        },
        body: JSON.stringify({
          client_id: env.GITHUB_CLIENT_ID,
          client_secret: env.GITHUB_CLIENT_SECRET,
          code: body.code,
        }),
      });

      const data = await response.json();
      return new Response(JSON.stringify(data), {
        headers: { "Content-Type": "application/json" },
      });
    }

    return new Response("Not Found", { status: 404 });
  },
};
```

`Services/AuthService.swift`:
```swift
import AuthenticationServices

actor AuthService {
    private let clientID = "YOUR_GITHUB_CLIENT_ID"
    private let proxyURL = "https://gitivity-oauth-proxy.YOUR_ACCOUNT.workers.dev"
    private let callbackScheme = "gitivity"
    private let keychain = KeychainService()

    func startOAuth(anchor: ASPresentationAnchor) async throws -> String {
        let authURL = URL(string: "https://github.com/login/oauth/authorize?client_id=\(clientID)&scope=repo&redirect_uri=gitivity://auth")!

        let code = try await withCheckedThrowingContinuation { (continuation: CheckedContinuation<String, Error>) in
            let session = ASWebAuthenticationSession(
                url: authURL,
                callbackURLScheme: callbackScheme
            ) { callbackURL, error in
                if let error {
                    continuation.resume(throwing: error)
                    return
                }
                guard let url = callbackURL,
                      let code = URLComponents(url: url, resolvingAgainstBaseURL: false)?
                        .queryItems?.first(where: { $0.name == "code" })?.value else {
                    continuation.resume(throwing: AuthError.noCode)
                    return
                }
                continuation.resume(returning: code)
            }
            session.presentationContextProvider = AnchorProvider(anchor: anchor)
            session.prefersEphemeralWebBrowserSession = true
            session.start()
        }

        let token = try await exchangeCode(code)
        try keychain.save(key: "github_token", value: token)
        return token
    }

    private func exchangeCode(_ code: String) async throws -> String {
        var request = URLRequest(url: URL(string: "\(proxyURL)/token")!)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["code": code])

        let (data, _) = try await URLSession.shared.data(for: request)
        let response = try JSONDecoder().decode(TokenResponse.self, from: data)
        return response.accessToken
    }

    func loadToken() -> String? {
        try? keychain.read(key: "github_token")
    }

    func logout() throws {
        try keychain.delete(key: "github_token")
    }
}

private struct TokenResponse: Decodable {
    let accessToken: String
    enum CodingKeys: String, CodingKey {
        case accessToken = "access_token"
    }
}

enum AuthError: Error {
    case noCode
    case tokenExchangeFailed
}

private class AnchorProvider: NSObject, ASWebAuthenticationPresentationContextProviding {
    let anchor: ASPresentationAnchor
    init(anchor: ASPresentationAnchor) { self.anchor = anchor }
    func presentationAnchor(for session: ASWebAuthenticationSession) -> ASPresentationAnchor {
        anchor
    }
}
```

```swift
import SwiftUI
import AuthenticationServices

@MainActor
final class AuthViewModel: ObservableObject {
    @Published var isAuthenticated = false
    @Published var isLoading = false
    @Published var error: String?

    private let authService = AuthService()

    init() {
        Task { await checkExistingToken() }
    }

    func checkExistingToken() async {
        isAuthenticated = await authService.loadToken() != nil
    }

    func login(anchor: ASPresentationAnchor) {
        isLoading = true
        error = nil
        Task {
            do {
                _ = try await authService.startOAuth(anchor: anchor)
                isAuthenticated = true
            } catch {
                self.error = error.localizedDescription
            }
            isLoading = false
        }
    }

    func logout() {
        try? Task { try await authService.logout() }
        isAuthenticated = false
    }
}
```

```swift
import Foundation

actor GitHubGraphQLService {
    private let endpoint = URL(string: "https://api.github.com/graphql")!
    private let keychain = KeychainService()

    func fetchContributions(from: Date? = nil, to: Date? = nil) async throws -> ContributionsCollection {
        let dateFormatter = ISO8601DateFormatter()
        var dateArgs = ""
        if let from, let to {
            dateArgs = "(from: \"\(dateFormatter.string(from: from))\", to: \"\(dateFormatter.string(from: to))\")"
        }

        let query = """
        {
          viewer {
            contributionsCollection\(dateArgs) {
              contributionCalendar {
                totalContributions
                weeks {
                  contributionDays {
                    contributionCount
                    date
                    color
                  }
                }
              }
              pullRequestContributions(first: 50) {
                nodes {
                  pullRequest {
                    title
                    body
                    url
                    state
                    createdAt
                    mergedAt
                    additions
                    deletions
                    changedFiles
                    repository { name owner { login } }
                  }
                }
              }
              issueContributions(first: 50) {
                nodes {
                  issue {
                    title
                    url
                    state
                    createdAt
                    closedAt
                    repository { name owner { login } }
                  }
                }
              }
              commitContributionsByRepository(maxRepositories: 20) {
                repository { name owner { login } }
                contributions { totalCount }
              }
            }
          }
        }
        """

        return try await executeQuery(query)
    }

    func fetchRecentCommits(owner: String, repo: String, since: Date) async throws -> [CommitEntry] {
        let dateFormatter = ISO8601DateFormatter()
        let query = """
        {
          repository(owner: "\(owner)", name: "\(repo)") {
            defaultBranchRef {
              target {
                ... on Commit {
                  history(first: 50, since: "\(dateFormatter.string(from: since))") {
                    nodes {
                      oid
                      message
                      committedDate
                      additions
                      deletions
                      repository: repository { name owner { login } }
                    }
                  }
                }
              }
            }
          }
        }
        """

        struct RepoData: Codable {
            let repository: RepoNode
        }
        struct RepoNode: Codable {
            let defaultBranchRef: BranchRef?
        }
        struct BranchRef: Codable {
            let target: TargetNode
        }
        struct TargetNode: Codable {
            let history: HistoryNode
        }
        struct HistoryNode: Codable {
            let nodes: [CommitEntry]
        }

        let response: GraphQLResponse<RepoData> = try await rawQuery(query)
        return response.data?.repository.defaultBranchRef?.target.history.nodes ?? []
    }

    private func executeQuery(_ query: String) async throws -> ContributionsCollection {
        struct ViewerData: Codable {
            let viewer: UserNode
        }
        let response: GraphQLResponse<ViewerData> = try await rawQuery(query)
        guard let data = response.data else {
            throw GitHubAPIError.noData(response.errors?.first?.message ?? "Unknown error")
        }
        return data.viewer.contributionsCollection
    }

    private func rawQuery<T: Codable>(_ query: String) async throws -> GraphQLResponse<T> {
        guard let token = try keychain.read(key: "github_token") else {
            throw GitHubAPIError.notAuthenticated
        }

        var request = URLRequest(url: endpoint)
        request.httpMethod = "POST"
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["query": query])

        let (data, _) = try await URLSession.shared.data(for: request)
        return try JSONDecoder().decode(GraphQLResponse<T>.self, from: data)
    }
}

enum GitHubAPIError: Error {
    case notAuthenticated
    case noData(String)
    case rateLimited
}
```

```swift
import Foundation

enum FeedGroupingService {
    private static let iso8601: ISO8601DateFormatter = {
        let f = ISO8601DateFormatter()
        f.formatOptions = [.withInternetDateTime]
        return f
    }()

    private static let groupingInterval: TimeInterval = 3600 // 1시간

    static func group(
        prs: [PullRequestEntry],
        commits: [CommitEntry],
        issues: [IssueEntry]
    ) -> [FeedItem] {
        var items: [FeedItem] = []

        // PR → 개별 FeedItem
        for pr in prs {
            items.append(FeedItem(
                id: pr.url,
                type: .pullRequest,
                title: pr.title,
                subtitle: "\(pr.repository.owner.login)/\(pr.repository.name)",
                repoName: pr.repository.name,
                repoOwner: pr.repository.owner.login,
                date: iso8601.date(from: pr.mergedAt ?? pr.createdAt) ?? .now,
                additions: pr.additions,
                deletions: pr.deletions,
                changedFiles: pr.changedFiles,
                pullRequest: pr,
                commits: []
            ))
        }

        // Commits → 레포 + 시간대별 그룹핑
        let commitsByRepo = Dictionary(grouping: commits) { $0.repository.name }
        for (repoName, repoCommits) in commitsByRepo {
            let sorted = repoCommits.sorted {
                ($0.committedDate) > ($1.committedDate)
            }

            var currentGroup: [CommitEntry] = []
            var groupStartDate: Date?

            for commit in sorted {
                let commitDate = iso8601.date(from: commit.committedDate) ?? .now

                if let start = groupStartDate,
                   abs(commitDate.timeIntervalSince(start)) > groupingInterval {
                    // 이전 그룹 마감
                    if !currentGroup.isEmpty {
                        items.append(makePushGroup(currentGroup))
                    }
                    currentGroup = [commit]
                    groupStartDate = commitDate
                } else {
                    currentGroup.append(commit)
                    if groupStartDate == nil { groupStartDate = commitDate }
                }
            }
            if !currentGroup.isEmpty {
                items.append(makePushGroup(currentGroup))
            }
        }

        // Issues → 개별 FeedItem
        for issue in issues {
            items.append(FeedItem(
                id: issue.url,
                type: .issue,
                title: issue.title,
                subtitle: "\(issue.repository.owner.login)/\(issue.repository.name)",
                repoName: issue.repository.name,
                repoOwner: issue.repository.owner.login,
                date: iso8601.date(from: issue.closedAt ?? issue.createdAt) ?? .now,
                additions: 0,
                deletions: 0,
                changedFiles: 0,
                pullRequest: nil,
                commits: []
            ))
        }

        return items.sorted { $0.date > $1.date }
    }

    private static func makePushGroup(_ commits: [CommitEntry]) -> FeedItem {
        let first = commits[0]
        let totalAdditions = commits.reduce(0) { $0 + $1.additions }
        let totalDeletions = commits.reduce(0) { $0 + $1.deletions }

        return FeedItem(
            id: "push-\(first.oid)-\(commits.count)",
            type: .pushGroup,
            title: "커밋 \(commits.count)회",
            subtitle: "\(first.repository.owner.login)/\(first.repository.name)",
            repoName: first.repository.name,
            repoOwner: first.repository.owner.login,
            date: iso8601.date(from: first.committedDate) ?? .now,
            additions: totalAdditions,
            deletions: totalDeletions,
            changedFiles: 0,
            pullRequest: nil,
            commits: commits
        )
    }
}
```

```swift
import Foundation

enum SummaryPeriod {
    case daily, weekly, monthly

    var label: String {
        switch self {
        case .daily: "일간"
        case .weekly: "주간"
        case .monthly: "월간"
        }
    }
}

enum PromptBuilder {
    static func buildFeedItemPrompt(_ item: FeedItem) -> String {
        var parts: [String] = []

        parts.append("다음 GitHub 활동에 대해 한국어로 간결하게 요약해주세요.")

        switch item.type {
        case .pullRequest:
            if let pr = item.pullRequest {
                parts.append("PR 제목: \(pr.title)")
                if let body = pr.body, !body.isEmpty {
                    parts.append("PR 설명: \(body)")
                }
                parts.append("상태: \(pr.state)")
                parts.append("변경: +\(pr.additions) -\(pr.deletions), \(pr.changedFiles)개 파일")
            }
        case .pushGroup:
            parts.append("커밋 \(item.commits.count)회 (\(item.repoOwner)/\(item.repoName))")
            for commit in item.commits.prefix(10) {
                parts.append("- \(commit.message)")
            }
            parts.append("총 변경: +\(item.additions) -\(item.deletions)")
        case .issue:
            parts.append("이슈: \(item.title)")
            parts.append("레포: \(item.repoOwner)/\(item.repoName)")
        }

        return parts.joined(separator: "\n")
    }

    static func buildPeriodSummaryPrompt(items: [FeedItem], period: SummaryPeriod) -> String {
        var parts: [String] = []

        parts.append("다음은 나의 GitHub \(period.label) 활동입니다. 한국어로 개발 회고 형식의 원페이저 요약을 작성해주세요.")
        parts.append("")

        let totalCommits = items.reduce(0) { $0 + (($1.type == .pushGroup) ? $1.commits.count : 0) }
        let prCount = items.filter { $0.type == .pullRequest }.count
        let issueCount = items.filter { $0.type == .issue }.count

        parts.append("총 커밋: \(totalCommits)회, PR: \(prCount)개, 이슈: \(issueCount)개")
        parts.append("")

        let repoGroups = Dictionary(grouping: items) { $0.repoName }
        for (repo, repoItems) in repoGroups {
            parts.append("[\(repo)]")
            for item in repoItems.prefix(5) {
                parts.append("- \(item.title) (+\(item.additions) -\(item.deletions))")
            }
        }

        return parts.joined(separator: "\n")
    }
}
```

```swift
import SwiftUI

@MainActor
final class FeedViewModel: ObservableObject {
    @Published var feedItems: [FeedItem] = []
    @Published var isLoading = false
    @Published var error: String?

    private let graphQLService = GitHubGraphQLService()
    private var lastFetchDate: Date?
    private let minimumFetchInterval: TimeInterval = 60 // 1분

    func loadFeed() async {
        if let last = lastFetchDate,
           Date().timeIntervalSince(last) < minimumFetchInterval {
            return
        }

        isLoading = true
        error = nil

        do {
            let contributions = try await graphQLService.fetchContributions()

            let prs = contributions.pullRequestContributions?.nodes?.map(\.pullRequest) ?? []
            let issues = contributions.issueContributions?.nodes?.map(\.issue) ?? []

            // 활발한 레포에서 커밋 가져오기
            var allCommits: [CommitEntry] = []
            let since = Calendar.current.date(byAdding: .day, value: -7, to: .now) ?? .now
            for repoContrib in (contributions.commitContributionsByRepository ?? []).prefix(5) {
                let commits = try await graphQLService.fetchRecentCommits(
                    owner: repoContrib.repository.owner.login,
                    repo: repoContrib.repository.name,
                    since: since
                )
                allCommits.append(contentsOf: commits)
            }

            feedItems = FeedGroupingService.group(prs: prs, commits: allCommits, issues: issues)
            lastFetchDate = .now
        } catch {
            self.error = error.localizedDescription
        }

        isLoading = false
    }
}
```

```swift
import Foundation

extension Date {
    var startOfWeek: Date {
        Calendar.current.dateInterval(of: .weekOfYear, for: self)?.start ?? self
    }

    var endOfWeek: Date {
        guard let start = Calendar.current.dateInterval(of: .weekOfYear, for: self)?.end else { return self }
        return Calendar.current.date(byAdding: .second, value: -1, to: start) ?? self
    }

    var startOfMonth: Date {
        Calendar.current.dateInterval(of: .month, for: self)?.start ?? self
    }

    var endOfMonth: Date {
        guard let start = Calendar.current.dateInterval(of: .month, for: self)?.end else { return self }
        return Calendar.current.date(byAdding: .second, value: -1, to: start) ?? self
    }

    var weekLabel: String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ko_KR")
        formatter.dateFormat = "M월 d일"
        let start = startOfWeek
        let end = endOfWeek
        return "\(formatter.string(from: start)) ~ \(formatter.string(from: end))"
    }
}
```

```swift
import SwiftUI

@MainActor
final class SummaryViewModel: ObservableObject {
    @Published var selectedPeriod: SummaryPeriod = .weekly
    @Published var contributionCalendar: ContributionCalendar?
    @Published var feedItems: [FeedItem] = []
    @Published var aiSummary: AISummary?
    @Published var isLoading = false
    @Published var isLoadingSummary = false
    @Published var error: String?
    @Published var currentDate = Date()
    @AppStorage("grassColorTheme") var selectedThemeID = "green"

    private let graphQLService = GitHubGraphQLService()
    private var aiProvider: AIProvider { AIProviderFactory.current }

    var selectedTheme: GrassColorTheme {
        GrassColorTheme.all.first { $0.id == selectedThemeID } ?? .green
    }

    var periodLabel: String {
        switch selectedPeriod {
        case .daily:
            currentDate.formatted(.dateTime.month().day().weekday())
        case .weekly:
            currentDate.weekLabel
        case .monthly:
            currentDate.formatted(.dateTime.year().month())
        }
    }

    func loadData() async {
        isLoading = true
        do {
            let (from, to) = dateRange(for: selectedPeriod, date: currentDate)
            let contributions = try await graphQLService.fetchContributions(from: from, to: to)

            contributionCalendar = contributions.contributionCalendar

            let prs = contributions.pullRequestContributions?.nodes?.map(\.pullRequest) ?? []
            let issues = contributions.issueContributions?.nodes?.map(\.issue) ?? []
            feedItems = FeedGroupingService.group(prs: prs, commits: [], issues: issues)
        } catch {
            self.error = error.localizedDescription
        }
        isLoading = false
    }

    func generateSummary() async {
        isLoadingSummary = true
        do {
            let prompt = PromptBuilder.buildPeriodSummaryPrompt(items: feedItems, period: selectedPeriod)
            aiSummary = try await aiProvider.summarize(prompt: prompt)
        } catch {
            self.error = error.localizedDescription
        }
        isLoadingSummary = false
    }

    func movePeriod(by offset: Int) {
        let calendar = Calendar.current
        switch selectedPeriod {
        case .daily:
            currentDate = calendar.date(byAdding: .day, value: offset, to: currentDate) ?? currentDate
        case .weekly:
            currentDate = calendar.date(byAdding: .weekOfYear, value: offset, to: currentDate) ?? currentDate
        case .monthly:
            currentDate = calendar.date(byAdding: .month, value: offset, to: currentDate) ?? currentDate
        }
        aiSummary = nil
        Task { await loadData() }
    }

    private func dateRange(for period: SummaryPeriod, date: Date) -> (Date, Date) {
        let calendar = Calendar.current
        switch period {
        case .daily:
            let start = calendar.startOfDay(for: date)
            let end = calendar.date(byAdding: .day, value: 1, to: start)!
            return (start, end)
        case .weekly:
            return (date.startOfWeek, date.endOfWeek)
        case .monthly:
            return (date.startOfMonth, date.endOfMonth)
        }
    }
}
```
